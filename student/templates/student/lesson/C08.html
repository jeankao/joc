# -*- coding: UTF-8 -*-
numbers = []

# 打開檔案
file = open("d:/numbers.txt", "r")

# 讀取檔案
for line in file.readlines():    #依序讀取每行
    line = line.strip()    #去掉每行頭尾空白
    for char in line :
        numbers.append(int(char))

# 關閉檔案
file.close()

print(numbers)
